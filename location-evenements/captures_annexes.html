from django.db import models

class Location(models.Model):
    STATUT_CHOICES = [
        ('en_attente', 'En attente'),
        ('confirmee', 'Confirmée'),
        ('en_cours', 'En cours'),
        ('terminee', 'Terminée'),
        ('annulee', 'Annulée'),
    ]
    client = models.ForeignKey('clients.Client', on_delete=models.PROTECT)
    type_evenement = models.ForeignKey('evenements.TypeEvenement', on_delete=models.SET_NULL, null=True)
    date_debut = models.DateField()
    date_fin = models.DateField()
    statut = models.CharField(max_length=20, choices=STATUT_CHOICES, default='en_attente')
    lieu_evenement = models.CharField(max_length=200)
    
    @property
    def duree_jours(self):
        return (self.date_fin - self.date_debut).days + 1

class LigneLocation(models.Model):
    location = models.ForeignKey(Location, related_name='lignes', on_delete=models.CASCADE)
    materiel = models.ForeignKey('materiels.Materiel', on_delete=models.PROTECT)
    quantite = models.PositiveIntegerField()
    prix_unitaire = models.DecimalField(max_digits=10, decimal_places=2)

2. Extrait de Code : Algorithme Anti-Chevauchement de Dates

locations/services.py
from django.db.models import Q
from .models import LigneLocation

def verifier_disponibilite(materiel_id, date_debut, date_fin, quantite_demandee):
    """
    Algorithme anti-chevauchement : Vérifie si la quantité de matériel 
    demandée est disponible pour la période spécifiée.
    """
    # 1. Identifier les locations qui se chevauchent avec la période demandée
    chevauchements = LigneLocation.objects.filter(
        materiel_id=materiel_id,
        location__statut__in=['confirmee', 'en_cours'],
        location__date_debut__lte=date_fin,
        location__date_fin__gte=date_debut
    )
    
    # 2. Calculer la quantité déjà réservée pendant cette période
    quantite_reservee = sum(ligne.quantite for ligne in chevauchements)
    
    # 3. Récupérer le stock total du matériel
    materiel = Materiel.objects.get(id=materiel_id)
    quantite_disponible = materiel.quantite_stock - quantite_reservee
    
    # 4. Vérifier si la quantité disponible est suffisante
    return quantite_disponible >= quantite_demandee
